# LARGEST NUMBET
a=int(input("enter the first number:"))
b=int(input("enter the second number:"))
c=int(input("enter the third number:"))
if a > b and a > c:
    print(a, "is the largest") 
elif b > a and b > c:
    print(b, "is the largest") 
else:
    print(c, "is the largest")

#SWAPPING NUMBERS

print("After swapping: a =", a, "b =", b)
